159  void calcDiscMatrix(double dt)
160  {
161  dt_ = dt;
162 
163  // Zero-order hold discretization
164  // See https://en.wikipedia.org/wiki/Discretization
165  if constexpr(StateDim == Eigen::Dynamic || InputDim == Eigen::Dynamic)
166  {
167  // If dimension is dynamic
168  if(E_.norm() == 0)
169  {
170  Eigen::MatrixXd ABZero(state_dim_ + input_dim_, state_dim_ + input_dim_);
171  ABZero << dt_ * A_, dt_ * B_, Eigen::MatrixXd::Zero(input_dim_, state_dim_ + input_dim_);
172  Eigen::MatrixXd ABZeroExp = ABZero.exp();
173  Ad_ = ABZeroExp.template block(0, 0, state_dim_, state_dim_);
174  Bd_ = ABZeroExp.template block(0, state_dim_, state_dim_, input_dim_);
175  Ed_.setZero(state_dim_);
176  }
177  else
178  {
179  // There is no proof that this is correct.
180  Eigen::MatrixXd ABEZero(state_dim_ + input_dim_ + 1, state_dim_ + input_dim_ + 1);
181  ABEZero << dt_ * A_, dt_ * B_, dt_ * E_, Eigen::MatrixXd::Zero(input_dim_ + 1, state_dim_ + input_dim_ + 1);
182  Eigen::MatrixXd ABEZeroExp = ABEZero.exp();
183  Ad_ = ABEZeroExp.template block(0, 0, state_dim_, state_dim_);
184  Bd_ = ABEZeroExp.template block(0, state_dim_, state_dim_, input_dim_);
185  Ed_ = ABEZeroExp.template block(0, state_dim_ + input_dim_, state_dim_, 1);
186  }
187  }
188  else
189  {
190  // If dimension is fixed
191  if(E_.norm() == 0)
192  {
193  Eigen::Matrix<double, StateDim + InputDim, StateDim + InputDim> ABZero;
194  ABZero << dt_ * A_, dt_ * B_, Eigen::Matrix<double, InputDim, StateDim + InputDim>::Zero();
195  Eigen::Matrix<double, StateDim + InputDim, StateDim + InputDim> ABZeroExp = ABZero.exp();
196  Ad_ = ABZeroExp.template block<StateDim, StateDim>(0, 0);
197  Bd_ = ABZeroExp.template block<StateDim, InputDim>(0, StateDim);
198  Ed_.setZero();
199  }
200  else

203  Eigen::Matrix<double, StateDim + InputDim + 1, StateDim + InputDim + 1> ABEZero;
