52  void calcGain(double horizon, double dt, OutputDimVector output_weight, InputDimVector input_weight)
53  {
54  if(horizon <= 0 || dt <= 0)
55  {
56  throw std::runtime_error("Input arguments of PreviewControl::calcGain are invalid. horizon: "
57  + std::to_string(horizon) + ", dt: " + std::to_string(dt));
58  }
59 
60  // 0. Calculate discrete system matrices
61  if(model_->dt_ != dt)
62  {
63  model_->calcDiscMatrix(dt);
64  }
65 
66  const Eigen::Matrix<double, StateDim, StateDim> & A = model_->Ad_;
67  const Eigen::Matrix<double, StateDim, InputDim> & B = model_->Bd_;
68  const Eigen::Matrix<double, OutputDim, StateDim> & C = model_->C_;
69 
70  Eigen::Matrix<double, OutputDim, OutputDim> Q = output_weight.asDiagonal();
71  Eigen::Matrix<double, InputDim, InputDim> R = input_weight.asDiagonal();
72  Eigen::Matrix<double, InputDim, InputDim> Rinv = input_weight.cwiseInverse().asDiagonal();
73 
74  // int stateDim = model_->stateDim();
75  int inputDim = model_->inputDim();
76  int outputDim = model_->outputDim();
77 
78  // 1. Calculate P by solving the Discrete-time Algebraic Riccati Equation
79  // (DARE) ref. https://scicomp.stackexchange.com/a/35394
80  int iterMax = 10000;
81  double relNormThre = 1e-8;
82  Eigen::Matrix<double, StateDim, StateDim> A0 = A;
83  Eigen::Matrix<double, StateDim, StateDim> G0 = B * Rinv * B.transpose();
84  Eigen::Matrix<double, StateDim, StateDim> H0 = C.transpose() * Q * C;
85  Eigen::Matrix<double, StateDim, StateDim> A1, G1, H1;
86  for(int i = 0; i < iterMax; i++)
87  {
88  Eigen::Matrix<double, StateDim, StateDim> I_GH_inv =
89  (Eigen::Matrix<double, StateDim, StateDim>::Identity() + G0 * H0).inverse();
90  A1 = A0 * I_GH_inv * A0;
91  G1 = G0 + A0 * I_GH_inv * G0 * A0.transpose();
92  H1 = H0 + A0.transpose() * H0 * I_GH_inv * A0;
93 
94  double relNorm = (H1 - H0).norm() / H1.norm();
95  if(relNorm < relNormThre)
96  {
97  break;
98  }
99  else if(i == iterMax - 1)
100  {
101  ROS_WARN_STREAM("[PreviewControl] Solution of Riccati equation did not converged: " << relNorm << " > "
102  << relNormThre);
103  }
104 
105  A0 = A1;
106  G0 = G1;
107  H0 = H1;
108  }
109  P_ = H1;
110 
111  riccati_error_ = (P_
112  - (A.transpose() * P_ * A + C.transpose() * Q * C
113  - A.transpose() * P_ * B * (R + B.transpose() * P_ * B).inverse() * B.transpose() * P_ * A))
114  .norm();
115 
116  // 2. Calculate the feedback gain (K and f)
117  preview_size_ = static_cast<int>(std::ceil(horizon / dt));
118  Eigen::Matrix<double, InputDim, InputDim> R_BtPB_inv = (R + B.transpose() * P_ * B).inverse();
119  // 2.1 Calculate K
120  K_ = R_BtPB_inv * B.transpose() * P_ * A;
121  // 2.2 Calculate f
122  F_.resize(inputDim, preview_size_ * outputDim);
123  Eigen::Matrix<double, StateDim, StateDim> A_BK = A - B * K_;
124  Eigen::Matrix<double, StateDim, StateDim> fSub = Eigen::Matrix<double, StateDim, StateDim>::Identity();
125  for(int i = 0; i < preview_size_; i++)
126  {
127  if(i < preview_size_ - 1)
128  {
129  F_.middleCols(i * outputDim, outputDim) = R_BtPB_inv * B.transpose() * fSub * C.transpose() * Q;
130  }
131  else
132  {
133  // See https://github.com/euslisp/jskeus/pull/551
134  F_.middleCols(i * outputDim, outputDim) = R_BtPB_inv * B.transpose() * fSub * P_ * C.transpose();
135  }
136  fSub = fSub * A_BK.transpose();
137  }
138  }
139

145  {
